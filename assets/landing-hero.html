Simple Configuration

okidoki.yaml

site:
    title: "{{site.title}}"
    description: "{{site.description}}"
    theme:
        light: "fantasy"
        dark: "forest"

globals:
    version: "{{globals.version}}"
    api_url: "https://api.example.com"

sidebars.yaml

menu:
    - title: Getting Started
        document: /index.md
    - title: Examples
        document: /markdown-examples.md
    - title: Demos
        document: /demo.md
